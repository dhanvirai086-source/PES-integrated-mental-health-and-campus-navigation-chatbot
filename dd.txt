operation = input("enter operation")
a = int(input("enter a num"))
b = int(input("enter a num"))
if operation == ("+"):
    sum = a+b
    print(sum)
elif operation == ("-"):
    sum = a-b
    print(sum)
elif operation == ("*"):
    sum = a*b
    print(sum)
elif operation == ("/"):
    sum = a/b
    print(sum)
elif operation == ("%"):
    sum = a%b
    print(sum)
else:
    print("invalid")
